## 1. prepare_visa.py 수정

### 배경

실제 VisA 데이터셋 배포 구조가 초기 설계와 달랐다:

| 항목 | 설계 기대 | 실제 |
|---|---|---|
| CSV 파일 구성 | 카테고리별 `candle.csv` … | 통합 `1cls.csv`, `2cls_fewshot.csv`, `2cls_highshot.csv` |
| 이미지 경로 컬럼 | `image_path` | `image` |
| 카테고리 식별 | CSV 파일명 | `object` 컬럼 |
| 이미지 경로 예시 | — | `candle/Data/Images/Normal/0836.JPG` |

### 수정 내용

| 함수/추가 | 변경 사항 |
|---|---|
| `reorganize_all()` | `1cls.csv` 존재 시 우선 사용; fallback → per-category CSV |
| `_reorganize_from_combined_csv()` | 신규 — `object` 컬럼으로 카테고리 그루핑 |
| `_process_rows()` | `image`/`image_path`, `mask`/`mask_path` 컬럼명 양쪽 지원 |
| `_process_category_task()` | 신규 — ProcessPoolExecutor pickle 호환 모듈 레벨 래퍼 |
| `--workers` | 신규 인자 — 카테고리 단위 병렬처리 (`run_parallel` 적용) |

#### Step 7-2a — Stage 1b ISP-AD

```python
import json, subprocess
from pathlib import Path
from concurrent.futures import ThreadPoolExecutor, as_completed
from tqdm.auto import tqdm

REPO = Path("/content/drive/MyDrive/project/aroma")
CONFIG = json.loads((REPO / "dataset_config.json").read_text())
NUM_WORKERS = 3

ISP_ENTRIES = [(k, v) for k, v in CONFIG.items()
               if not k.startswith("_") and v["domain"] == "isp"]

def process_entry(key, entry):
    seed_dir = Path(entry["seed_dir"])
    seeds = sorted(seed_dir.glob("*.png")) if seed_dir.exists() else []
    if not seeds:
        return key, 0, 0
    cat_dir = seed_dir.parents[1]
    done = 0
    for seed in seeds:
        out = cat_dir / "stage1b_output" / seed.stem
        if (out / "seed_profile.json").exists():
            done += 1; continue
        subprocess.run(
            ["python", "stage1b_seed_characterization.py",
             "--seed_defect", str(seed), "--output_dir", str(out)],
            cwd=REPO, check=True, capture_output=True
        )
        done += 1
    return key, len(seeds), done

with ThreadPoolExecutor(max_workers=NUM_WORKERS) as ex:
    futures = {ex.submit(process_entry, k, v): k for k, v in ISP_ENTRIES}
    for fut in tqdm(as_completed(futures), total=len(ISP_ENTRIES), desc="Stage1b ISP-AD"):
        key, total, done = fut.result()
        print(f"  ✓ {key}: {done}/{total} seeds")
print("✓ ISP-AD Stage 1b 완료")
```

#### Step 7-3 — Stage 1b subtype 분포 검증

```python
import json
from pathlib import Path
from collections import Counter
from concurrent.futures import ThreadPoolExecutor, as_completed
from tqdm.auto import tqdm

BASE = Path("/content/drive/MyDrive/data/Aroma")
DATASETS = (
    [BASE / f"isp/unsupervised/{m}"   for m in ["ASM","LSM_1","LSM_2"]] +
    [BASE / f"mvtec/{c}"              for c in
     ["bottle","cable","capsule","carpet","grid","hazelnut","leather",
      "metal_nut","pill","screw","tile","toothbrush","transistor","wood","zipper"]] +
    [BASE / f"visa/{c}"               for c in
     ["candle","capsules","cashew","chewinggum","fryum",
      "macaroni1","macaroni2","pcb1","pcb2","pcb3","pcb4","pipe_fryum"]]
)
NUM_WORKERS = 8

def check_dataset(cat_dir):
    profiles = list((cat_dir / "stage1b_output").rglob("seed_profile.json")) \
               if (cat_dir / "stage1b_output").exists() else []
    if not profiles:
        return cat_dir, [], None
    dist = Counter(json.loads(p.read_text())["subtype"] for p in profiles)
    return cat_dir, profiles, dist

results = {}
with ThreadPoolExecutor(max_workers=NUM_WORKERS) as ex:
    futures = {ex.submit(check_dataset, d): d for d in DATASETS}
    for fut in tqdm(as_completed(futures), total=len(DATASETS), desc="subtype 검증"):
        cat_dir, profiles, dist = fut.result()
        results[cat_dir] = (profiles, dist)

print(f"{'Dataset':20s}  {'seeds':>6}  subtype distribution")
print("-" * 65)
all_ok = True
for cat_dir in DATASETS:
    profiles, dist = results[cat_dir]
    if dist is None:
        print(f"{cat_dir.name:20s}  [MISSING]"); all_ok = False; continue
    dist_str = "  ".join(f"{k}:{v}" for k, v in dist.most_common())
    print(f"{cat_dir.name:20s}  {len(profiles):>6}  {dist_str}")
```

```python
import json
from pathlib import Path
from concurrent.futures import ThreadPoolExecutor, as_completed
from tqdm.auto import tqdm

BASE = Path("/content/drive/MyDrive/data/Aroma")
DATASETS = (
    [BASE / f"isp/unsupervised/{m}"   for m in ["ASM","LSM_1","LSM_2"]] +
    [BASE / f"mvtec/{c}"              for c in
     ["bottle","cable","capsule","carpet","grid","hazelnut","leather",
      "metal_nut","pill","screw","tile","toothbrush","transistor","wood","zipper"]] +
    [BASE / f"visa/{c}"               for c in
     ["candle","capsules","cashew","chewinggum","fryum",
      "macaroni1","macaroni2","pcb1","pcb2","pcb3","pcb4","pipe_fryum"]]
)
NUM_WORKERS = 8

def check_stage2(cat_dir):
    stage2_dir = cat_dir / "stage2_output"
    if not stage2_dir.exists():
        return cat_dir, 0, 0, False
    seed_dirs  = [d for d in stage2_dir.iterdir() if d.is_dir()]
    total_vars = sum(len(list(d.glob("*.png"))) for d in seed_dirs)
    return cat_dir, len(seed_dirs), total_vars, True

results = {}
with ThreadPoolExecutor(max_workers=NUM_WORKERS) as ex:
    futures = {ex.submit(check_stage2, d): d for d in DATASETS}
    for fut in tqdm(as_completed(futures), total=len(DATASETS), desc="variant 검증"):
        cat_dir, n_seeds, n_vars, exists = fut.result()
        results[cat_dir] = (n_seeds, n_vars, exists)

print(f"{'Dataset':20s}  {'seeds':>6}  {'variants':>9}")
print("-" * 42)
all_ok = True
for cat_dir in DATASETS:
    n_seeds, n_vars, exists = results[cat_dir]
    if not exists:
        print(f"{cat_dir.name:20s}  [MISSING]"); all_ok = False; continue
    print(f"{cat_dir.name:20s}  {n_seeds:>6}  {n_vars:>9}")
print("\n" + ("✓ Stage 3 진행 가능" if all_ok else "✗ MISSING 항목 확인"))
```

```python
CONFIG = json.loads((REPO / "dataset_config.json").read_text())
REQUIRED_KEYS = {"defect_path", "x", "y", "scale", "rotation",
                 "suitability_score", "matched_background_type"}

# cat_dir 목록 (중복 제거, domain 순 정렬)
seen, cat_dirs = set(), []
for k, v in CONFIG.items():
    if k.startswith("_"):
        continue
    cat_dir = Path(v["seed_dir"]).parents[1]
    key = (v["domain"], str(cat_dir))
    if key not in seen:
        seen.add(key)
        cat_dirs.append((v["domain"], cat_dir))
cat_dirs.sort()

def check_stage3(cat_dir):
    stage3_dir = cat_dir / "stage3_output"
    if not stage3_dir.exists():
        return 0, 0, "MISSING"
    seed_dirs = [d for d in stage3_dir.iterdir() if d.is_dir()]
    total, schema_ok = 0, True
    for sd in seed_dirs:
        pmap = sd / "placement_map.json"
        if not pmap.exists():
            schema_ok = False
            continue
        entries = json.loads(pmap.read_text())
        total += len(entries)
        for e in entries:
            for placement in e.get("placements", []):
                if not REQUIRED_KEYS.issubset(placement.keys()):
                    schema_ok = False
    return len(seed_dirs), total, ("✓" if schema_ok else "✗ schema 오류")

print(f"{'Domain':6s}  {'Dataset':20s}  {'seeds':>6}  {'placements':>11}  {'schema':>12}")
print("-" * 62)
all_ok = True
for domain, cat_dir in tqdm(cat_dirs, desc="placement 검증"):
    n_seeds, n_placements, status = check_stage3(cat_dir)
    if status == "MISSING":
        print(f"{domain:6s}  {cat_dir.name:20s}  [MISSING]")
        all_ok = False
    else:
        print(f"{domain:6s}  {cat_dir.name:20s}  {n_seeds:>6}  {n_placements:>11}  {status:>12}")
        if status != "✓":
            all_ok = False

print("\n" + ("✓ Stage 4 진행 가능" if all_ok else "✗ MISSING 또는 schema 오류 확인"))
```

#### Step 10-3 — Stage 4 합성 결과 검증

```python
import json
from pathlib import Path
from concurrent.futures import ThreadPoolExecutor, as_completed
from tqdm.auto import tqdm

BASE = Path("/content/drive/MyDrive/data/Aroma")
DATASETS = (
    [BASE / f"isp/unsupervised/{m}"   for m in ["ASM","LSM_1","LSM_2"]] +
    [BASE / f"mvtec/{c}"              for c in
     ["bottle","cable","capsule","carpet","grid","hazelnut","leather",
      "metal_nut","pill","screw","tile","toothbrush","transistor","wood","zipper"]] +
    [BASE / f"visa/{c}"               for c in
     ["candle","capsules","cashew","chewinggum","fryum",
      "macaroni1","macaroni2","pcb1","pcb2","pcb3","pcb4","pipe_fryum"]]
)
NUM_WORKERS = 8

def check_stage4(cat_dir):
    stage4_dir = cat_dir / "stage4_output"
    if not stage4_dir.exists():
        return cat_dir, 0, 0, False
    seed_dirs = [d for d in stage4_dir.iterdir() if d.is_dir()]
    total = sum(len(list((sd / "defect").glob("*.png"))) for sd in seed_dirs)
    return cat_dir, len(seed_dirs), total, True

results = {}
with ThreadPoolExecutor(max_workers=NUM_WORKERS) as ex:
    futures = {ex.submit(check_stage4, d): d for d in DATASETS}
    for fut in tqdm(as_completed(futures), total=len(DATASETS), desc="합성 결과 검증"):
        cat_dir, n_seeds, n_images, exists = fut.result()
        results[cat_dir] = (n_seeds, n_images, exists)

print(f"{'Dataset':20s}  {'seeds':>6}  {'images':>8}")
print("-" * 40)
all_ok = True
for cat_dir in DATASETS:
    n_seeds, n_images, exists = results[cat_dir]
    if not exists:
        print(f"{cat_dir.name:20s}  [MISSING]"); all_ok = False; continue
    print(f"{cat_dir.name:20s}  {n_seeds:>6}  {n_images:>8}")
print("\n" + ("✓ 전체 완료" if all_ok else "✗ MISSING 항목 확인"))
```
